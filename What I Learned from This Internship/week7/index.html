def _initialize_centroids(self, X):
    centroids = np.zeros((self.n_clusters, X.shape[1]))
    # First centroid: random
    centroids[0] = X[np.random.randint(len(X))]
    
    for i in range(1, self.n_clusters):
        distances = np.zeros(len(X))
        for j in range(len(X)):
            min_dist = np.inf
            for k in range(i):
                dist = np.sum((X[j] - centroids[k]) ** 2)
                if dist < min_dist:
                    min_dist = dist
            distances[j] = min_dist
        
        # Probability proportional to distance¬≤
        probs = distances ** 2 / np.sum(distances ** 2)
        next_idx = np.random.choice(len(X), p=probs)
        centroids[i] = X[next_idx]
    return centroids
Convergence Check Tolerance-based stopping 
def fit(self, X):
    self.centroids = self._initialize_centroids(X)
    
    for iteration in range(self.max_iter):
        distances = self._calculate_distance(X)
        self.labels = self._assign_clusters(distances)
        
        new_centroids = self._update_centroids(X, self.labels)
        
        # Check convergence
        centroid_shift = np.sqrt(np.sum(
            (new_centroids - self.centroids) ** 2, 
            axis=1
        )).max()
        
        if centroid_shift < self.tol:
            print(f"Converged at iteration {iteration}")
            break
            
        self.centroids = new_centroids
