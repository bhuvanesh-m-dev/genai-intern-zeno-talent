01. The Engine: KMeans Class

Instead of just using Scikit-Learn, I implemented the K-Means clustering logic from scratch to understand the math behind centroid convergence and inertia calculation.

KMeans.py 
class KMeans:
    def __init__(self, n_clusters=3, max_iter=300):
        self.n_clusters = n_clusters
        self.max_iter = max_iter
        self.centroids = None

    def fit(self, X):
        # 1. Initialize Centroids
        self.centroids = self._initialize_centroids(X)

        for _ in range(self.max_iter):
            # 2. Assign clusters
            distances = self._calc_distance(X, self.centroids)
            self.labels = self._assign_clusters(distances)

            # 3. Update centroids
            new_centroids = self._update_centroids(X, self.labels)
            
            # 4. Check convergence
            if np.all(self.centroids == new_centroids):
                break
            self.centroids = new_centroids
